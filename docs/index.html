Стэк технологий

Мониторинг в реальном времени

  • Проблема потока
    WebSocket блокирует главный поток
  • Асинхронная архитектура
    Использование реактивного подхода со StateFlow в связке с корутинами; сетевой парсинг JSON вынесен в Dispatchers.IO
  • Результаты
    Плавность интерфейса сохраняется при массовом спаме логов

Обработка ошибок

  • Жизненный цикл Android
    Обычный StateFlow кэширует состояние, из-за чего при повороте экрана пользователя может выкинуть на экран логина повторно
  • Паттерн "Fire-and-Forget"
    Для инвалидации сессии применен SharedFlow; cобытие транслируется один раз и не сохраняется в памяти
  • Изоляция слоев
    ViewModel перехватывает ошибку сети и лишь "дергает за ниточку"; слой UI реагирует на это логаутом, оставаясь полностью отвязанным от сетевой логики.

Кастомный UI

SQLite


                // db.DatabaseHelper
                fun favoriteUpsert(command: String) {
                    //insert new favorite command or update last used timestamp of existing one

                    val db = writableDatabase
                    val currentTime = System.currentTimeMillis()

                    //check exists
                    val cursor = db.rawQuery(
                        "SELECT id FROM favorites WHERE commandText = ?",
                        arrayOf(command)
                    )
                    if (cursor.moveToFirst()) {
                        //update time
                        val id = cursor.getInt(0)
                        val values = ContentValues().apply {
                            put("lastUsed", currentTime)
                        }
                        db.update(
                            "favorites",
                            values,
                            "id = ?",
                            arrayOf(id.toString())
                        )
                    } else {
                        val values = ContentValues().apply {
                            put("commandText", command)
                            put("lastUsed", currentTime)
                        }
                        db.insert("favorites", null, values)
                    }

                    cursor.close()
                }

Работа в фоне


                // MainActivity
                private fun startBackgroundMonitoring() {
                    val workRequest = PeriodicWorkRequestBuilder<PcMonitorWorker>(
                        15,
                        TimeUnit.MINUTES
                    ).build()
                    val workManager = WorkManager.getInstance(this)
                    workManager.enqueueUniquePeriodicWork(
                        "pc_monitor",
                        ExistingPeriodicWorkPolicy.KEEP,
                        workRequest
                    )

                    workManager.getWorkInfosForUniqueWorkLiveData("pc_monitor")
                        .observe(this) { workInfos ->
                            if (workInfos.isNotEmpty()) {
                                val isAnyFinished = workInfos.any {
                                    it.state.isFinished
                                }
                                if (isAnyFinished) {
                                    viewModel.historyRefresh()
                                }
                            }
                        }
                }
